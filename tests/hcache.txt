	set header_cache_backend = lmdb

Script to run all the combinations

#!/bin/bash

for i in mutt*; do
	echo "========================================================================="
	echo $i
	for j in *.rc; do
		echo "-----------------"
		echo $j
		./$i -n -F $j -v | grep "^hcache"
		$i -n -F $j -D | grep hcache_backend
		echo
	done
done >& m2
